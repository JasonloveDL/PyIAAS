124    @staticmethod
125    def default_sample_strategy(original_size, new_size):
126        """
127        default sample strategy described in paper
128        :param original_size: original layer width
129        :param new_size: new layer width, we have new_size >original_size
130        :return: sampling sequence
131        """
132        seq = list(range(original_size))
133        num_to_sample = new_size - original_size
134        while num_to_sample != 0:
135            sample_number = min(original_size, num_to_sample)
136            seq += random.sample(seq, sample_number)
137            num_to_sample = num_to_sample - sample_number
138        return seq

576        self.Uc = nn.Parameter(torch.empty(hidden_size, hidden_size))
577        self.Uo = nn.Parameter(torch.empty(hidden_size, hidden_size))

579        self.bi = nn.Parameter(torch.zeros(hidden_size))

594    def forward_step(self, x, h, c):
595        """
596        do one step LSTM forward calculation
597        @param x: input shape [batch, feature]
598        """
599        ft = torch.sigmoid(torch.matmul(x, self.Wf) + torch.matmul(h, self.Uf) + self.bf)
600        it = torch.sigmoid(torch.matmul(x, self.Wi) + torch.matmul(h, self.Ui) + self.bi)
601        ot = torch.matmul(x, self.Wo) + torch.matmul(h, self.Uo) + self.bo
602        c_ = torch.tanh(torch.matmul(x, self.Wc) + torch.matmul(h, self.Uc) + self.bc)
603        c = ft * c + it * c_
604        h = ot * torch.tanh(c) + ot
605        return h, c

608        '''
609        forward propagation
610        @param x:  input of size [batch, feature, seqlen]
611        @return: hidden state (batch, feature,seqlen)
612        '''
613        h = torch.zeros((x.shape[0], self.hidden_size), device=x.device)
614        c = torch.zeros((x.shape[0], self.hidden_size), device=x.device)
615        output = []
616        for i in range(x.shape[2]):
617            h, c = self.forward_step(x[:, :, i], h, c)
618            output.append((h, c))
619        output = [torch.stack([i[0] for i in output]), torch.stack([i[1] for i in output])]
620        return output[0].permute(1, 2, 0)
